PRESS ENTER SELECT FIRST OPTION AND THE DOUBLE CLICK IT THE ENTER THE FILE NAME IN THE FILE OPENED

DECLARE
	LN_COUNT NUMBER;
BEGIN

	FND_STANDARD.SET_WHO;
	
	SELECT COUNT(*) INTO LN_COUNT 
	FROM DEPARTMENT_AD 
	WHERE UPPER(:DEPARTMENT_AD.DEPT_NAME) = UPPER(DEPT_NAME);
	IF LN_COUNT>0 THEN
		FND_MESSAGE.SET_STRING('DEPARTMENT '||:DEPARTMENT_AD.DEPT_NAME||'ALREADY EXISTS!');
		FND_MESSAGE.SHOW;
	END IF;
END;
